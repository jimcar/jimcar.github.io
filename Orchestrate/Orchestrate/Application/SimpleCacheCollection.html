def save(data)
  key, ref = data.key, data.ref
  @primary_keys[key] = PrimaryKey.new(ref) if primary_keys[key].nil?
  if data.respond_to? :etype and data.etype
    @primary_keys[key].add_event(data.etype, ref)
  else
    @primary_keys[key].ref = ref
  end
end
